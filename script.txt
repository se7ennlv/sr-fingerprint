==> Delete Fingerprint duplicate records (Optional)
WITH cte AS (
SELECT 
WUserID, 
Years,
Months, 
Days, 
Hours, 
Minutes, 
Seconds, 
DevID,
LocationID,

ROW_NUMBER() OVER(
PARTITION BY WUserID, Years, Months, Days, Hours, Minutes, Seconds, DevID, LocationID
ORDER BY WUserID, Years, Months, Days, Hours, Minutes, Seconds, DevID,LocationID
) row_num
FROM UserLogInfo
)

SELECT *
FROM cte
WHERE row_num > 1
